% other
\newcommand{\addnoiseplot}[3]{\addplot+[
    color=#3,
    mark=*,
    mark options={scale=1},
    error bars/error bar style={thick}
    ] table[
    col sep=comma,
    x=noise, 
    y expr=(\thisrow{#2_med} * 100),
    y error minus expr=((\thisrow{#2_med} - \thisrow{#2_q1}) * 100),
    y error plus expr=((\thisrow{#2_q3} - \thisrow{#2_med}) * 100)
    ] {#1};}
\newcommand{\addsizeplot}[3]{\addplot+[
    color=#3,
    mark=*,
    mark options={scale=1},
    error bars/error bar style={thick}
    ] table[
    col sep=comma,
    x=n, 
    y expr=(\thisrow{opt_med_#2} * 100),
    y error minus expr=((\thisrow{opt_med_#2} - \thisrow{opt_q1_#2}) * 100),
    y error plus expr=((\thisrow{opt_q3_#2} - \thisrow{opt_med_#2}) * 100)
    ] {#1};}
